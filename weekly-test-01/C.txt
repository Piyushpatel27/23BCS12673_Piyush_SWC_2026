select * from Submissions;

select distinct likes.parent_id as post_id ,count(*) as number_of_comments
from (select sub_id from Submissions where parent_id is NULL) as likes
join (select sub_id from Submissions where parent_id is not NULL) as comment
on likes.parent_id= comment.parent_id
group by comment.parent_id
order by likes.parent_id;
